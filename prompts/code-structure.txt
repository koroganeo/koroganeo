```typescript
export class ExcelService {
  private metadata: Map<string, ArticleMetadata> = new Map();
  
  async loadMetadata(filePath: string): Promise<void> {
    const workbook = new ExcelJS.Workbook();
    await workbook.xlsx.readFile(filePath);
    const sheet = workbook.worksheets[0];
    
    sheet.eachRow((row, index) => {
      if (index === 1) return; // Skip header
      
      const metadata: ArticleMetadata = {
        genres: row.getCell(1).value as string,
        titleVi: row.getCell(2).value as string,
        titleEn: row.getCell(3).value as string,
        tags: this.parseArray(row.getCell(4).value as string),
        slug: row.getCell(5).value as string,
        page: row.getCell(6).value as number,
        difficultyLevel: row.getCell(7).value as string,
        creators: this.parseArray(row.getCell(8).value as string),
        createdAt: new Date(row.getCell(9).value as string),
        crawlStatus: row.getCell(10).value as string,
        length: row.getCell(11).value as number,
      };
      
      this.metadata.set(this.getSlugKey(metadata.slug), metadata);
    });
  }
  
  private parseArray(value: string): string[] {
    // Parse "['item1', 'item2']" format
    return JSON.parse(value.replace(/'/g, '"'));
  }
  
  private getSlugKey(url: string): string {
    // Extract filename from slug URL
    return url.split('/').pop() || '';
  }
}
```

2. CACHE SERVICE (Performance for 646 articles):
```typescript
export class CacheService {
  private articleCache: Map<string, { content: string; metadata: ArticleMetadata }> = new Map();
  private searchIndex: Map<string, Set<string>> = new Map(); // word -> slugs
  
  async warmUp(excelService: ExcelService, wordService: WordService): Promise<void> {
    // Load all metadata first
    const allMetadata = excelService.getAllMetadata();
    
    // Process Word files in batches of 50 to avoid memory issues
    for (let i = 0; i < allMetadata.length; i += 50) {
      const batch = allMetadata.slice(i, i + 50);
      await Promise.all(batch.map(meta => this.cacheArticle(meta, wordService)));
    }
  }
  
  async cacheArticle(metadata: ArticleMetadata, wordService: WordService): Promise<void> {
    const content = await wordService.parseWordFile(metadata.slug);
    this.articleCache.set(metadata.slug, { content, metadata });
    this.indexForSearch(metadata.slug, metadata, content);
  }
  
  private indexForSearch(slug: string, metadata: ArticleMetadata, content: string): void {
    const words = this.tokenize(metadata.titleVi + ' ' + metadata.titleEn + ' ' + content);
    words.forEach(word => {
      if (!this.searchIndex.has(word)) this.searchIndex.set(word, new Set());
      this.searchIndex.get(word)!.add(slug);
    });
  }
  
  private tokenize(text: string): string[] {
    return normalizeVietnamese(text.toLowerCase())
      .split(/\s+/)
      .filter(w => w.length > 2);
  }
}
```

```typescript
export function normalizeVietnamese(text: string): string {
  // Remove Vietnamese diacritics for search
  const diacritics: Record<string, string> = {
    'à|á|ạ|ả|ã|â|ầ|ấ|ậ|ẩ|ẫ|ă|ằ|ắ|ặ|ẳ|ẵ': 'a',
    'è|é|ẹ|ẻ|ẽ|ê|ề|ế|ệ|ể|ễ': 'e',
    'ì|í|ị|ỉ|ĩ': 'i',
    'ò|ó|ọ|ỏ|õ|ô|ồ|ố|ộ|ổ|ỗ|ơ|ờ|ớ|ợ|ở|ỡ': 'o',
    'ù|ú|ụ|ủ|ũ|ư|ừ|ứ|ự|ử|ữ': 'u',
    'ỳ|ý|ỵ|ỷ|ỹ': 'y',
    'đ': 'd',
  };
  
  Object.keys(diacritics).forEach(key => {
    text = text.replace(new RegExp(key, 'g'), diacritics[key]);
  });
  
  return text;
}
```

```typescript
import { Injectable, signal, computed, linkedSignal, resource } from '@angular/core';
import { HttpClient } from '@angular/common/http';

@Injectable({ providedIn: 'root' })
export class ArticleService {
  private http = inject(HttpClient);
  
  // Input signals
  currentPage = signal(1);
  selectedGenre = signal<string | null>(null);
  selectedTags = signal<string[]>([]);
  selectedCreators = signal<string[]>([]);
  difficulty = signal<string | null>(null);
  sortBy = signal<'date' | 'length' | 'title'>('date');
  sortOrder = signal<'asc' | 'desc'>('desc');
  
  // LinkedSignal for language (persisted to localStorage)
  language = linkedSignal(() => {
    const stored = localStorage.getItem('lang');
    return (stored === 'en' ? 'en' : 'vi') as 'vi' | 'en';
  });
  
  // Resource API for async data loading (Angular 21)
  articles = resource({
    request: () => ({
      page: this.currentPage(),
      genre: this.selectedGenre(),
      tags: this.selectedTags(),
      creators: this.selectedCreators(),
      difficulty: this.difficulty(),
      sortBy: this.sortBy(),
      sortOrder: this.sortOrder(),
      lang: this.language(),
    }),
    loader: async ({ request }) => {
      const params = new URLSearchParams();
      Object.entries(request).forEach(([key, value]) => {
        if (value !== null && value !== undefined) {
          if (Array.isArray(value)) {
            value.forEach(v => params.append(key, v));
          } else {
            params.append(key, String(value));
          }
        }
      });
      
      const response = await fetch(`/api/articles?${params}`);
      return response.json();
    },
  });
  
  // Computed signals
  hasActiveFilters = computed(() => 
    this.selectedGenre() !== null || 
    this.selectedTags().length > 0 || 
    this.selectedCreators().length > 0 || 
    this.difficulty() !== null
  );
  
  totalPages = computed(() => {
    const data = this.articles.value();
    return data ? Math.ceil(data.total / 20) : 0;
  });
  
  // Effects
  constructor() {
    effect(() => {
      const lang = this.language();
      localStorage.setItem('lang', lang);
      document.documentElement.lang = lang;
    });
  }
  
  // Methods
  clearFilters() {
    this.selectedGenre.set(null);
    this.selectedTags.set([]);
    this.selectedCreators.set([]);
    this.difficulty.set(null);
  }
  
  toggleTag(tag: string) {
    this.selectedTags.update(tags => 
      tags.includes(tag) 
        ? tags.filter(t => t !== tag)
        : [...tags, tag]
    );
  }
}
```

3. SEARCH SERVICE (Debounced with Signals):
```typescript
import { Injectable, signal, computed, effect } from '@angular/core';
import { debounceTime, distinctUntilChanged } from 'rxjs';
import { toObservable, toSignal } from '@angular/core/rxjs-interop';

@Injectable({ providedIn: 'root' })
export class SearchService {
  private http = inject(HttpClient);
  
  searchTerm = signal('');
  isSearching = signal(false);
  
  // Convert signal to observable for RxJS operators
  private searchTerm$ = toObservable(this.searchTerm).pipe(
    debounceTime(300),
    distinctUntilChanged()
  );
  
  // Convert back to signal
  debouncedTerm = toSignal(this.searchTerm$, { initialValue: '' });
  
  // Search results
  results = computed(() => {
    const term = this.debouncedTerm();
    if (!term || term.length < 2) return [];
    
    // Trigger search effect
    this.performSearch(term);
    return this.cachedResults();
  });
  
  private cachedResults = signal<Article[]>([]);
  
  private async performSearch(term: string) {
    this.isSearching.set(true);
    try {
      const results = await this.http.get<{ results: Article[] }>(
        `/api/search?q=${encodeURIComponent(term)}&lang=${this.languageService.current()}`
      ).toPromise();
      this.cachedResults.set(results?.results || []);
    } finally {
      this.isSearching.set(false);
    }
  }
}
```
